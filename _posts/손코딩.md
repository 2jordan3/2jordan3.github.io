
## 문자열 역순
```C
//C언어
int strrev(char *src)
{
  char *end_pointer = NULL;

  if (!src) return -1;
  if (!*src) return -1;

  end_pointer = src + strlen(src) - 1;

  while (src < end_pointer) {
    char tmp = *src;
    *src = *end_pointer;
    *end_pointer = tmp;
    src++;
    end_pointer--;
  }

  return 0;
}
```

## 이진탐색
```C
//C언어
int binary_search(int array[], int size, int value)
{
  int first = 0;
  int last = size - 1;
  int mid = 0;
  int i = 0;

  if (!array) return -1;
  if (size < 0) return -1;

  while (first != last) {
    mid = (first + last) / 2;
    if (array[mid] > value) {
      last = mid - 1;
    } else if (array[mid] < value) {
      first = mid + 1;
    } else {
      return mid;
    }
  }

  if (array[first] == value) return first;

  return -1;
}
```
